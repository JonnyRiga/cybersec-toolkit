# SHELL FUNCTIONS
www()    { python3 -m http.server ${1:-80}; }          # serve current dir; optional port arg
ips()    { ip -4 -brief addr show; }                   # quick view of all IPv4 interfaces
mkcd()   { mkdir -p "$1" && cd "$1"; }                 # make dir and cd into it
docs()   { cat ~/Tools/Documents/"$@"; }               # read from ~/Tools/Documents/
cheats() { cat ~/Tools/Documents/Cheatsheets/"$@"; }   # read from ~/Tools/Documents/Cheatsheets/

nmapz() {  # full port scan with -sC -sV -O; usage: nmapz <target> [outname] [T3|T4|T5] [-- flags]
  local target="$1" out="${2:-nmap_scan}" timing="T4"
  shift 2 2>/dev/null
  [[ "$1" =~ ^T[3-5]$ ]] && { timing="$1"; shift; }
  [ "$1" = "--" ] && shift
  mkdir -p nmap
  nmap "$target" -p- -oA "nmap/$out" -sC -sV -O -"$timing" "$@"
}

xmas() {   # Xmas scan; usage: xmas <target> [extra nmap flags]
  local target="$1"; shift
  nmap -sX -T4 -n -Pn -p- "$target" "$@"
}

reconweb() {  # scaffold a recon directory; usage: reconweb <name> [target]
  # creates: exploits/ loot/ screenshots/ notes/ scans/{nmap,gobuster,nikto,whatweb,ffuf}
  local name="$1" target="${2:-}"
  mkdir -p "$name"/{exploits,loot,screenshots,notes}
  mkdir -p "$name"/scans/{nmap,gobuster,nikto,whatweb,ffuf}
  [[ -n "$target" ]] && echo "$target" > "$name/target"
  cd "$name"
}
